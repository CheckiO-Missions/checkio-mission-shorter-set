Here it is: 
def remove_min_max(data: set, total:int) -> set:
    
    if len(data) != 0: # the best way - to use just 'if data:'
        order = sorted(data)
        for i in range(total):
            data.discard(order[i])
            data.discard(order[-i-1])
            
    return data
